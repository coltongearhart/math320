        \item Find the pdf using the cdf method.\vspace{200pt}
    \end{enumerate}
    
    \item Let $X \follow{Geometric}(p = 0.3)$ and $Y = \sqrt{X}$. Find the pmf of $Y$, $f_Y(y)$.\vspace{200pt}% Original
